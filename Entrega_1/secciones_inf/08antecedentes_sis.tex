\subsection{Zona Sísmica y parámetros}

\begin{itemize}
    \item Zona Sísmica: 3
    \item Categoría ocupación: II
    \item Tipo de suelo: A
    \item Aceleración efectiva másxima del suelo: $A_0 = 0.4g$
    \item Coeficiente de importancia: $I = 1$
    \item $S=9$, $T_0=0.15s$, $T'=0.2s$, $n=1$, $p=2$
    \item Coeficiente sísmico: $C = \frac{2.75SA_0}{gR}(\frac{T'}{T*})^n$
    \item Coeficiente sísmico mínimo: $C_{min} = \frac{A_0S}{6g} = 0.06$
    \item Corte basal: $Q_0 = C_{min}IP = 0.06P$
    \item Factores de modificación de respuesta (Muros HA): $R=7$ y $R_0=11$
    \item Coeficiente sísmico máximo: $C_{max} = \frac{0.35SA_0}{g} = 0.126$
\end{itemize}
